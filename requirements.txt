# ---------------- БИБЛИОТЕКИ ДЛЯ ВЕБ-ПРИЛОЖЕНИЯ ----------------

Flask==3.0.3
# Flask — основной веб-фреймворк.
# Он позволяет:
#  - поднимать веб-сервер,
#  - обрабатывать запросы (адреса /login, /tickets и т.п.),
#  - возвращать HTML-страницы и JSON.
# Версия 3.0.3 — актуальный стабильный релиз.

Flask-Login==0.6.3
# Flask-Login — расширение для Flask,
# которое отвечает за:
#  - вход пользователя в систему,
#  - выход,
#  - хранение информации о том, кто сейчас залогинен (current_user),
#  - защиту маршрутов через @login_required.
# Без него нам пришлось бы самим писать всю авторизацию.

Flask-SQLAlchemy==3.1.1
# Flask-SQLAlchemy — обёртка над библиотекой SQLAlchemy,
# которая позволяет:
#  - описывать таблицы базы данных как Python-классы (User, Ticket),
#  - легко делать запросы к БД (User.query.filter_by(...)),
#  - не писать вручную SQL-запросы.
# Это делает работу с БД проще и безопаснее.

Flask-Bcrypt==1.0.1
# Flask-Bcrypt — библиотека для безопасного хранения паролей.
# Она:
#  - не хранит пароли в открытом виде,
#  - превращает пароль в "хэш" с помощью алгоритма bcrypt,
#  - позволяет проверять пароль через user.check_password(...).
# Это важно для безопасности: даже если базу украдут,
# настоящих паролей там нет.

SQLAlchemy==2.0.29
# SQLAlchemy — "двигатель" для работы с базой данных.
# Flask-SQLAlchemy использует его под капотом.
# Именно он превращает методы вроде User.query.filter_by(...)
# в реальные SQL-запросы к базе (SQLite).


# ----------------- БИБЛИОТЕКИ ДЛЯ ТЕСТОВ И ПРОВЕРОК -----------------

pytest==8.2.1
# pytest — фреймворк для автоматического тестирования.
# Мы используем его для:
#  - запуска тестов из файла tests/test_api.py,
#  - проверки, что API и логика приложения работают правильно.
# При ошибке в логике тесты падают — это сигнал, что что-то сломалось.

pytest-cov==5.0.0
# pytest-cov — расширение для pytest,
# которое измеряет "покрытие кода тестами".
# Показывает, какая часть нашего кода реально проверяется тестами,
# а какая ни разу не выполнялась.
# Это полезно для отчёта и для качества проекта.

bandit==1.7.9
# Bandit — инструмент для проверки безопасности кода.
# Он просматривает Python-файлы и ищет:
#  - небезопасные конструкции,
#  - возможные уязвимости (например, "захардкоженные" пароли),
#  - подозрительные места с точки зрения безопасности.
# В нашем GitHub Actions он запускается автоматически,
# и "ругается", если находит потенциальную проблему.
